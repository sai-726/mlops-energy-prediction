# Quick GitHub Push Commands
# Run these commands in PowerShell AFTER restarting PowerShell

# Navigate to project
cd c:\Users\krish\Desktop\Kiran-MLops

# Verify Git is installed
git --version

# Configure Git (replace with YOUR name and email)
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# Initialize Git repository
git init

# Check what files will be added
git status

# Add all files
git add .

# IMPORTANT: Verify .env is NOT in the list
git status

# If you see .env in the list, remove it:
# git reset .env

# Commit your code
git commit -m "Initial commit: Complete MLOps energy prediction project"

# Now go to GitHub.com and create a new repository
# Then come back and run these commands (replace YOUR_USERNAME and REPO_NAME):

# Add GitHub remote
git remote add origin https://github.com/YOUR_USERNAME/REPO_NAME.git

# Push to GitHub
git branch -M main
git push -u origin main

# You'll be asked for:
# Username: Your GitHub username
# Password: Use a Personal Access Token (create at https://github.com/settings/tokens)
